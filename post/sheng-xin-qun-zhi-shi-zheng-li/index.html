homer

#可以根据一段DNA序列预测其转录因子在bioconda中搜索homer，可以看到有很多种版本
conda search homer -c bioconda

#默认安装 homer 最新版
conda install -c bioconda homer -y # -y 表示直接安装，不询问确认与否
